$ pkg update && pkg upgrade

$ pkg install python3

$ pkg install git

$ pip install requests

$ git clone https://github.com/Rizki-Xs/spamming

$ cd spamming

$ ls

$ pip install -r requirements.txt

$ python3 R_run.py
